SQL para criação do Banco de Dados:
```sql
CREATE TABLE `suahora`.`salas`
(`id` INT NOT NULL AUTO_INCREMENT , `nome` VARCHAR(255) NOT NULL , `capacidade` INT(40) NOT NULL , `tamanho` INT(40) NOT NULL , PRIMARY KEY (`id`))
ENGINE = InnoDB;
```
